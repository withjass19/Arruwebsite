22. How to convert boolean to String

Code


public class BooleanToStringConversion {
    public static void main(String[] args) {
        // Example boolean values
        boolean trueValue = true;
        boolean falseValue = false;

        // Using String.valueOf()
        String stringValue1 = String.valueOf(trueValue);
        String stringValue2 = String.valueOf(falseValue);

        // Using concatenation with an empty string
        String stringValue3 = trueValue + "";
        String stringValue4 = falseValue + "";

        // Display the results
        System.out.println("Using String.valueOf():");
        System.out.println("trueValue: " + stringValue1);
        System.out.println("falseValue: " + stringValue2);

        System.out.println("\nUsing concatenation with an empty string:");
        System.out.println("trueValue: " + stringValue3);
        System.out.println("falseValue: " + stringValue4);
    }
}

Output

Using String.valueOf():
trueValue: true
falseValue: false

Using concatenation with an empty string:
trueValue: true
falseValue: false
